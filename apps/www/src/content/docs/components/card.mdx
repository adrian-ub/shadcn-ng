## Usage

```ts
import {
  UbCardContentDirective,
  UbCardDescriptionDirective,
  UbCardDirective,
  UbCardFooterDirective,
  UbCardHeaderDirective,
  UbCardTitleDirective,
} from '~/components/ui/card'
```

```html
 <div ubCard>
    <div ubCardHeader>
        <h3 ubCardTitle>Card Title</h3>
        <p ubCardDescription>Card Description</p>
    </div>
    <div ubCardContent>
       <p>Card Content</p>
    </div>
    <div ubCardFooter>
        <p>Card Footer</p>
    </div>
</div>
```
